# Panduan Game Cyber Security

Tiap LEVEL berbeda-beda dalam mencari Password untuk lanjut ke LEVEL berikutnya, maka, Lu harus cari cara bagaimana tiap LEVEL ketemu passwordnya.

# Command yang bisa lu pakai

ls, ls -la, cd, cat, find, base64, strings, grep, echo, sort, uniq, tr, cut, xargs

# Installation ( Termux )

1. pkg update
2. pkg upgrade -y
3. pkg install git -y
4. pkg install bash -y
5. pkg install coreutils -y
6. pkg install util-linux -y
7. pkg install ncurses-utils -y
8. git clone https://github.com/Boiescylosh/cyber
9. cd cyber
10. chmod +x cyber.sh
11. ./cyber.sh atau bash cyber.sh

# Installation ( Linux )

1. sudo apt update
2. sudo apt upgrade -y
3. sudo apt install git -y
4. sudo apt install bash -y
5. sudo apt install coreutils -y
6. sudo apt install util-linux -y
7. sudo apt install ncurses-bin -y
8. git clone https://github.com/Boiescylosh/cyber
9. cd cyber
10. chmod +x cyber.sh
11. ./cyber.sh atau bash cyber.sh

# Installation ( WSL )

sudo apt update && sudo apt install git bash coreutils util-linux ncurses-bin -y
